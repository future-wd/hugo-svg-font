{{- /* HUGO SVG FONT usage: {{ partial "scripts/svg-font" (dict "src" "icons/bootstrap/envelope" "em" 1 "block" true/false )}} */ -}} {{- $random := delimit (shuffle (split (md5 "salt") "" )) "" -}} {{- /* init variables with defaults */ -}} {{- $src := "" -}} {{- $em := 1 -}} {{- $block := false -}} {{- $title := "" -}} {{- $desc := "" -}} {{- /* if dict passed as context to partial */ -}} {{- if reflect.IsMap . -}} {{- $src = .src -}} {{- with .title -}} {{- $title = . -}} {{- end -}} {{- with .desc -}} {{- $desc = . -}} {{- end -}} {{- with .em -}} {{- $em := . -}} {{- end -}} {{- with .block -}} {{- $block = true -}} {{- end -}} {{- else -}} {{- /* if called with {{ partial "svg-font" "/icons/name.svg" }} */ -}} {{- $src = . -}} {{- end -}} {{- /* add svg file suffix if not present in $src */ -}} {{- if not (strings.HasSuffix $src ".svg") -}} {{- $src = printf "%s.svg" $src -}} {{- end -}} {{- /* get svg file from resources folder */ -}} {{- $svgFile := resources.Get $src -}} {{- /* extract text components from svg file with regex */ -}} {{- $height := index (findRE `height=\"([0-9]*)\"` $svgFile.Content 1) 0 -}} {{- $width := index (findRE `width=\"([0-9]*)\"` $svgFile.Content 1) 0 -}} {{- $viewBox := index (findRE `viewBox=\"([0-9\s]*)\"` $svgFile.Content 1) 0 -}} {{- $paths := delimit (findRE `` $svgFile.Content) "" -}} {{- /* format title */ -}} {{- $ariaTitle := "" -}} {{- with $title -}} {{- $titleId := printf "%s-title" $random }} {{- $title = printf `%s` $titleId . -}} {{- $ariaTitle = printf `aria-labelledby="%s"` $titleId -}} {{- end -}} {{- /* format desc */ -}} {{- $ariaDesc := "" -}} {{- with $desc -}} {{- $descId := printf "%s-desc" $random }} {{- $desc = printf `%s` $descId . -}} {{- $ariaDesc = printf `aria-describedby="%s"` $descId -}} {{- end -}} {{- /* assign classes */ -}} {{- $class := (slice "svg-font" (printf "em-%v" $em)) -}} {{- if $block -}} {{- $class = $class | append "d-block" -}} {{- end -}} {{- $class = delimit $class " " -}} {{- $class = printf `class="%s"` $class -}} {{- $attr := (slice $height $width $viewBox $class $ariaTitle $ariaDesc) -}} {{- $content := (slice $title $desc $paths) -}} {{- $svgModified := printf `` (delimit $attr " ") (delimit $content " ") -}} {{- $svgModified | safeHTML -}}
